90 void EPoller::updateChannel(Channel* channel){
91  assertInLoopThread();
92  LOG_TRACE << "fd = " << channel->fd() << " events = " << channel->events();
93  const int index = channel->index();
94  if(index == kNew || index == kDeleted){
95  // a new one, add with EPOLL_CTL_ADD
96  int fd = channel->fd();
97  if(index == kNew){
98  assert(channels_.find(fd) == channels_.end());
99  channels_[fd] = channel;
100  }else{ // index == kDeleted
101  assert(channels_.find(fd) != channels_.end());
102  assert(channels_[fd] == channel);
103  }
104  channel->set_index(kAdded);
105  update(EPOLL_CTL_ADD, channel);
106  }else{
107  // update existing one with EPOLL_CTL_MOD/DEL
108  int fd = channel->fd();
109  (void)fd;
110  assert(channels_.find(fd) != channels_.end());
111  assert(channels_[fd] == channel);
112  assert(index == kAdded);
113  if(channel->isNoneEvent()){
114  update(EPOLL_CTL_DEL, channel);
115  channel->set_index(kDeleted);
116  }else{
117  update(EPOLL_CTL_MOD, channel);
118  }
119  }

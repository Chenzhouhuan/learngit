51  loop_->assertInLoopThread();
52  ssize_t nwrote = 0;
53  // if no thing in output queue, try writing directly
54  if(!channel_->isWriting() && outputBuffer_.readableBytes() == 0){
55  nwrote = ::write(channel_->fd(), message.data(), message.size());
56  if(nwrote >= 0){
57  if(implicit_cast<size_t>(nwrote) < message.size()){
58  LOG_TRACE << "I am going to write more data";
59  }else if(writeCompleteCallback_){

61  }
62  }else{

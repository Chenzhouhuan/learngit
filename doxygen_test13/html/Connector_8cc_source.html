48 void Connector::connect(){
49  int sockfd = sockets::createNonblockingOrDie();
50  int ret = sockets::connect(sockfd, serverAddr_.getSockAddrInet());
51  int savedErrno = (ret == 0) ? 0 : errno;
52  switch(savedErrno){
53  case 0:
54  case EINPROGRESS: //代表连接还在进行中
55  case EINTR: //系统调用被中断
56  case EISCONN: //表示连接已经建立
57  connecting(sockfd);
58  break;
59 
60  case EAGAIN: //再尝试一次(例如连续做read操作而没有数据可读，此时程序不会阻塞等待，而是放回该信号)
61  case EADDRINUSE: //监听的端口已被使用
62  case EADDRNOTAVAIL: //本地端口不够用
63  case ECONNREFUSED: //被服务器拒绝
64  case ENETUNREACH: //当前无法从本主机访问网络
65  retry(sockfd);
66  break;
67 
68  case EACCES: //没有权限操作
69  case EPERM: //没有权限操作
70  case EAFNOSUPPORT: //远程端点指定的地址族不能与这种类型的套接字一起使用
71  case EALREADY: //超时，过时
72  case EBADF: //无效描述符
73  case EFAULT: //说明在对套接字上调用某些函数的时候传入了空指针等非法參数
74  case ENOTSOCK: //表示sockfd是一个文件描述符而不是socket描述符
75  LOG_SYSERR << "connect error in Connector::startInLoop " << savedErrno;
76  sockets::close(sockfd);
77  break;
78 
79  default:
80  LOG_SYSERR << "Unexpected error in Connector::startInLoop " << savedErrno;
81  sockets::close(sockfd);
82  break;
83  }
84 }

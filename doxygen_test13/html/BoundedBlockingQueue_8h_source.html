18 template<typename T>
19 class BoundedBlockingQueue : noncopyable
20 {
21  public:
22  explicit BoundedBlockingQueue(int maxSize)
23  : mutex_(),
24  notEmpty_(mutex_),
25  notFull_(mutex_),
26  queue_(maxSize)
27  {
28  }
29 
30  void put(const T& x)
31  {
32  MutexLockGuard lock(mutex_);
33  while (queue_.full())
34  {
35  notFull_.wait();
36  }
37  assert(!queue_.full());
38  queue_.push_back(x);
39  notEmpty_.notify();
40  }
41 
42  void put(T&& x)
43  {
44  MutexLockGuard lock(mutex_);
45  while (queue_.full())
46  {
47  notFull_.wait();
48  }
49  assert(!queue_.full());
50  queue_.push_back(std::move(x));
51  notEmpty_.notify();
52  }
53 
54  T take()
55  {
56  MutexLockGuard lock(mutex_);
57  while (queue_.empty())
58  {
59  notEmpty_.wait();
60  }
61  assert(!queue_.empty());
62  T front(std::move(queue_.front()));
63  queue_.pop_front();
64  notFull_.notify();
65  return front;
66  }
67 
68  bool empty() const
69  {
70  MutexLockGuard lock(mutex_);
71  return queue_.empty();
72  }
73

75  {

78  }

81  {

87  {

89  return queue_.capacity();
90  }

92  private:
93  mutable MutexLock mutex_;
94  Condition notEmpty_ GUARDED_BY(mutex_);
95  Condition notFull_ GUARDED_BY(mutex_);
96  boost::circular_buffer<T> queue_ GUARDED_BY(mutex_);
97 };

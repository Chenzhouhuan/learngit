18 template<typename T>
19 class BlockingQueue : noncopyable

21  public:

29  void put(const T& x)
30  {
31  MutexLockGuard lock(mutex_);
32  queue_.push_back(x);
33  notEmpty_.notify(); // wait morphing saves us
34  // http://www.domaigne.com/blog/computing/condvars-signal-with-mutex-locked-or-not/
35  }

37  void put(T&& x)
38  {
39  MutexLockGuard lock(mutex_);
40  queue_.push_back(std::move(x));
41  notEmpty_.notify();
42  }

44  T take()
45  {
46  MutexLockGuard lock(mutex_);
47  // always use a while-loop, due to spurious wakeup
48  while (queue_.empty())
49  {
50  notEmpty_.wait();
51  }
52  assert(!queue_.empty());
53  T front(std::move(queue_.front()));
54  queue_.pop_front();
55  return front;
56  }
57 
58  size_t size() const
59  {
60  MutexLockGuard lock(mutex_);
61  return queue_.size();
62  }
63 
64  private:
65  mutable MutexLock mutex_;
66  Condition notEmpty_ GUARDED_BY(mutex_);
67  std::deque<T> queue_ GUARDED_BY(mutex_);
68 };
